# ─────────────────────────────────────────────────────────────
#  zapret-gui — встроенные стратегии (defaults)
#
#  Эти стратегии — готовые конфигурации nfqws2 с фильтрами.
#  Содержат полную командную строку: --filter-* + --lua-desync=*
#  Секции внутри одной стратегии разделены --new.
#
#  Протокол определяется автоматически из --filter-tcp / --filter-udp.
# ─────────────────────────────────────────────────────────────

[tcp_default]
name = Default — базовая стратегия
author = zapret-gui
label = recommended
description = Стандартная стратегия: fake + multisplit для HTTP/TLS, fake для QUIC
--filter-tcp=80
--filter-l7=http
--payload=http_req
--lua-desync=fake:blob=fake_default_http
--lua-desync=multisplit:pos=method+2
--new
--filter-tcp=443
--filter-l7=tls
--payload=tls_client_hello
--lua-desync=fake:blob=fake_default_tls
--lua-desync=multisplit:pos=1,midsld
--new
--filter-udp=443
--filter-l7=quic
--payload=quic_initial
--lua-desync=fake:blob=fake_default_quic:repeats=6

[tcp_hostfake]
name = HostFake — подмена SNI в fake-пакетах
author = zapret-gui
label = stable
